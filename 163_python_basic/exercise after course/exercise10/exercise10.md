from cffi import FFI

def cal(a, b):
    ffi = FFI()

    c_a = ffi.cast('int', a)
    c_b = ffi.cast('int', b)
    
    ffi.cdef('''
        int  foo(int a, int b);
    ''')

    cdll = ffi.dlopen("a.dll")
    c = cdll.foo(c_a, c_b)
    return c

if __name__ == "__main__":
    print(cal(1.23, 4.56))
Python变量到C变量转换使用ffi.cast()函数，声明C中函数使用ffi.cdef()，加载动态链接库使用ffi.dlopen()。

    c_a = ffi.new('int[]', len(a))
    ffi.memmove(c_a, a, ffi.sizeof(c_a))

    ffi.cdef('''
        int  *foo(int *a);
    ''')

    cdll = ffi.dlopen("a.dll")
    c_b = cdll.foo(c_a)
    return ffi.unpack(c_b, ffi.sizeof(c_a))

if __name__ == "__main__":

新建变量使用ffi.new()，拷贝数组使用ffi.memmove()，声明C中函数使用ffi.cdef()，加载动态链接库使用ffi.dlopen()，将数组转换成列表使用ffi.unpack()。
