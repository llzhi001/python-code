class Echo():
    def __init__(self, name):
        self.name = name
        print("Hello {}!".format(name))
    def __add__(self, others):
        if self.name == others.name:
            print("Hello, the SAME name!".format(self.name))
        else:
            print("Hello, NOT the same!")
    
s = input()
a = 0
for i in s:
    if 'A' <= i <= 'Z':
        a += 1
echoA = Echo(s)
echoB = Echo(a)
