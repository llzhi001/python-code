#include <Python.h>
int main()
{    
    char fname[] = "mxmul.py";
    FILE *fp = fopen(fname, "r");
    if (fp == NULL)
    {        
        printf("fopen error");
        return 1;
    }    
    
    Py_Initialize();
    PyRun_SimpleFile(fp, fname);
    Py_Finalize();
    
    return 0;
