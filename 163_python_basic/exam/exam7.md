class Creature:
    count = 0
    def __init__(self, weight, long, width, height):
        self.wt = weight
        self.lo = long
        self.wi = width
        self.he = height
        Creature.count += 1
    
    def getDensity(self):
        vol = (3.14159*pow(max(self.lo, self.wi)/2,2)*self.he)
        return self.wt/vol

avg_density = 0
try:
    s = input()
    while s != "":
        wt, lo, wi, he = tuple(s.split())
        d = Creature(eval(wt), eval(lo), eval(wi), eval(he))
        avg_density += d.getDensity()
        s = input()
except:
    pass
avg_density =  avg_density / Creature.count
print("{:.2f}".format(avg_density))
